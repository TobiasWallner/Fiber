1004 template<std::unsigned_integral UInt>
1005 OStream& operator<<(OStream& stream, FormatHex<UInt> hex){
1006 char buffer[sizeof(hex)*2];
1007 char* itr = &buffer[0];
1008
1009 if(hex.include_header){
1010 *itr++ = '0';
1011 *itr++ = 'x';
1012 }
1013
1014 // print at least 1 zero
1015 if(hex._value == 0 && !hex.force_digits){
1016 *itr++ = '0';
1017 return stream << FormatStr::like(buffer, itr, hex);
1018 }
1019
1020 int i = 0;
1021
1022 // optional skipp zeros
1023 if(!hex.force_digits){
1024 for(; i < 8; ++i){
1025 uint32_t h = (hex._value >> (4 * (7 - i))) & 0xF;
1026 if(h != 0){
1027 break;
1028 }
1029 }
1030 }
1031
1032 // hex tables
1033 constexpr char hex_upper[] = {'1', '2', '3', '4', '5', '6', '7', '8', '9', 'A', 'B', 'C', 'D', 'E', 'F'};
1034 constexpr char hex_lower[] = {'1', '2', '3', '4', '5', '6', '7', '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'};
1035
1036 // print hex characters
1037 for(; i < 8; ++i){
1038 uint32_t h = (hex._value >> (4 * (7 - i))) & 0xF;
1039 *itr++ = (hex.lowercase) ? hex_lower[h] : hex_upper[h];
1040 }
1041
1042 // output the formated string
1043 return stream << FormatStr::like(buffer, itr, hex);
