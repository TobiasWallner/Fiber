221 void promote(){
222 // promote await back into running queue
223 for(TaskBase* task : this->_await_bench){
224 if(!task->is_awaiting()){
225 logger::log_move(this->now(), task->name(), task->id(), "await", "run");
226 this->running_queue().push(task);
227 }
228 }
229 this->_await_bench.erase_if([](const TaskBase* task){return !task->is_awaiting();});
230
231 // promote waiting queue into running queue
232 while(!this->waiting_queue().empty()){
233 TaskBase* task = this->waiting_queue().top();
234 const TimePoint now = this->now();
235 if(task->ready_time() <= now || task->immediatelly_ready()){
236 logger::log_move(now, task->name(), task->id(), "wait", "run");
237 this->waiting_queue().pop();
238 this->running_queue().push(task);
239 }else{
240 return;
241 }
242 }
243 }

248 void sleep(){

266 void run_next(){
267 TaskBase* task = this->running_queue().top_pop();
268 task->_execution_start = this->now();
269 fiber::detail::frame_allocator = task->_frame_allocator;
270 task->resume();
271 logger::log_resume(task->_execution_start, this->now(), task->name(), task->id());
272 // re-schedule
273
274 const CoSignal signal = task->get_signal();
275 switch(signal.type()){
276 case CoSignal::Type::Await : {
277 // clear signal
278 this->_await_bench.emplace_back(task);
279 logger::log_move(this->now(), task->name(), task->id(), "resume", "await");
280 }break;
281 case CoSignal::Type::NextCycle : {
282 task->_schedule = task->next_schedule(task->_schedule, ExecutionTime{task->_execution_start, this->now()});
283 this->waiting_queue().push(task);
284 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
285 }break;
286 case CoSignal::Type::ImplicitDelay : {
287 const Duration rel_deadline = task->_schedule.deadline - task->_schedule.ready;
288 task->_schedule.ready = this->now() + fiber::rounding_duration_cast<Duration>(signal.delay());
289 task->_schedule.deadline = task->_schedule.ready + rel_deadline;
290 this->waiting_queue().push(task);
291 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
292 }break;
293 case CoSignal::Type::ExplicitDelay : {
294 task->_schedule.ready = this->now() + fiber::rounding_duration_cast<Duration>(signal.delay());
295 task->_schedule.deadline = task->_schedule.ready + fiber::rounding_duration_cast<Duration>(signal.deadline());
296 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
297 }break;
298 case CoSignal::Type::None : {
299 /*
300 - Nothing to do
301 - task probably finished
302 - let it die by not inserting it into lists
303 */
304 logger::log_delete(this->now(), task->name(), task->id());
305 } break;
306 default : {
307 /*
308 - Nothing to do
309 - task probably finished
310 - let it die by not inserting it into lists
311 */
312 logger::log_delete(this->now(), task->name(), task->id());
313 } break;
314 }
315 }
