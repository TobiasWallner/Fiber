63 inline void GPIO_config_pin_output(GPIO_TypeDef* GPIOx, uint32_t pin_mask){
64 // read current config
65 uint32_t reg = GPIOx->MODER;
66
67 // calculate the position of the corresponding pin bit pattern
68 const uint32_t pos = pin_mask * pin_mask;
69
70 // clear section
71 reg &= ~((pos << 1) | pos);
72
73 // set to output
74 reg |= (pos);
75
