265 void on_ready(void (*callback)(const T&)){
266 // if the future is already ready and the promise was kept --> call the callback directly
267 if(this->is_ready()){
268 // call the callback
269 callback(this->get());
270 this->set_state(State::BrokenPromise); // for consistency
271 }else{
272 // get lock
273 this->acquire_dual_locks();
274
275 // double check - an interrupt might have fired between the first check and the lock
276 if(this->is_ready()){
277 // release lock in controll path 1/2
278 this->release_dual_locks();
279
280 // call the callback
281 callback(this->get());
282 this->set_state(State::BrokenPromise); // for consistency
283 }else{
284 // safely install the callback function and detatch the future from the promise
285 if(this->_promisePtr == nullptr){
286 EMBED_THROW(Exception("Assigned callback to detatched future."));
287 }else if(this->_promisePtr->_callback != nullptr){
288 EMBED_THROW(Exception("Already assigned a callback to the future promise pair."));
289 }else{
