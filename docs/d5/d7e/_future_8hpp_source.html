618 void acquire_dual_locks(){
619 // early return
620 if(this->_futurePtr == nullptr){return;}
621
622 // disable interrupts to prevent deadlocks
623 fiber::disable_interrupts();
624
625 #ifdef FIBER_MULTI_CORE
626
627
628 while(true){
629 // lock its own lock
630 while (this->_lock.exchange(Lock::Closed, std::memory_order_acquire) != Lock::Open) {/* spin */}
631
632 // double check - because there might have been an interrupt before the first check and the disabling of the interrupts
633 if(this->_futurePtr == nullptr){
634 // cleanup if value changed
635
636 // open lock
637 this->_lock.store(Lock::Open, std::memory_order_release);
638
639 // enable interrupts
640 fiber::enable_interrupts();
641
642 return;
643 }
644
645 // try to lock the other lock once, if failed, open both locks so that the future can get both locks, then re-try
646 if(this->_futurePtr->_lock.exchange(Future<T>::Lock::Closed, std::memory_order_acquire) != Future<T>::Lock::Open){
647 // release the lock so that the future may acquire both locks
648 this->_lock.store(Lock::Open, std::memory_order_release);
649
650 // failed to get the lock --> contine loop and retry
651 continue;
652 }else{
653 // succeeded in acquireing both locks --> stop trying and exit loop
654 break;
655 }
656 }
