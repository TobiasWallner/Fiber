55 template<std::unsigned_integral T>
56 constexpr ClockTick(const T v){
57 // try as much as possible to avoid '%'
58 if constexpr (std::numeric_limits<T>::max() <= max_tick){
59 this->value = static_cast<UInt>(v);
60 } else if constexpr (modulo_is_power_of_two){
61 this->value = static_cast<UInt>(v & static_cast<T>(max_tick));
62 } else if constexpr (std::numeric_limits<T>::max()/2 <= max_tick){
63 this->value = static_cast<UInt>((v < modulo) ? v : v-modulo);
64 }else if (v <= max_tick) {
65 this->value = static_cast<UInt>(v);
66 }else{
67 this->value = static_cast<UInt>(v % modulo);
68 }
69 }

104 friend constexpr ClockTick operator+(const ClockTick lhs, const ClockTick rhs){
105 if constexpr (max_tick_is_limit){
106 // if the overflow happens at the maximal representable value: just add them and let them wrap around by themselfes
107 return ClockTick::reinterpret(static_cast<UInt>(lhs.value + rhs.value));
108 }else if constexpr (modulo_is_power_of_two){
109 // use a mask to do a quick modulo
110 return ClockTick::reinterpret(static_cast<UInt>((lhs.value + rhs.value) & max_tick));
111 }else{
112 // if all else failes, fall back

115 }
