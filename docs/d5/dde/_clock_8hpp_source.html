157 template<std::unsigned_integral T>
158 constexpr ClockTick(const T v){
159 // try as much as possible to avoid '%'
160 if constexpr (std::numeric_limits<T>::max() <= max_tick){
161 this->value = static_cast<UInt>(v);
162 } else if constexpr (modulo_is_power_of_two){
163 this->value = static_cast<UInt>(v & static_cast<T>(max_tick));
164 } else if constexpr (std::numeric_limits<T>::max()/2 <= max_tick){
165 this->value = static_cast<UInt>((v < modulo) ? v : v-modulo);
166 }else if (v <= max_tick) {
167 this->value = static_cast<UInt>(v);
168 }else{
169 this->value = static_cast<UInt>(v % modulo);
170 }
171 }

173 template<std::signed_integral T>
174 constexpr ClockTick(const T v){

176 *this = ClockTick(static_cast<UInt>(v));

206 friend constexpr ClockTick operator+(const ClockTick lhs, const ClockTick rhs){
207 if constexpr (max_tick_is_limit){
208 // if the overflow happens at the maximal representable value: just add them and let them wrap around by themselfes
209 return ClockTick::reinterpret(static_cast<UInt>(lhs.value + rhs.value));
210 }else if constexpr (modulo_is_power_of_two){
211 // use a mask to do a quick modulo
212 return ClockTick::reinterpret(static_cast<UInt>((lhs.value + rhs.value) & max_tick));
213 }else{
214 // if all else failes, fall back
215 const auto result = lhs.value + rhs.value;
216 return ClockTick::reinterpret(static_cast<UInt>((result < modulo) ? result : result - modulo));
217 }
218 }

227 friend constexpr ClockTick operator-(ClockTick lhs, ClockTick rhs){
228 if constexpr (max_tick_is_limit){
229 // if the overflow happens at the maximal representable value: just add them and let them wrap around by themselfes
230 lhs.value -= rhs.value;
231 return lhs;
232 }else {
233 // fallback
234 const auto result = lhs.value - rhs.value;
235 return ClockTick::reinterpret(static_cast<UInt>((lhs.value < rhs.value) ? result + modulo : result));
236 }

251 friend constexpr ClockTick operator-(const ClockTick v){

272 return ClockTick::reinterpret(static_cast<UInt>(lhs.value * rhs.value));
