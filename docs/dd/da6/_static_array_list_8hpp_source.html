466 template<class Callable>
467 std::size_t erase_if(Callable&& f){
468 iterator read_itr = this->begin();
469 const iterator end_itr = this->end();
470
471 // search first that is true
472 for(;read_itr != end_itr; ++read_itr){
473 if(f(*read_itr)) break;
474 }
475
476 // return if it did not aplie to any
477 if(read_itr == end_itr) return 0;
478
479 // setup first read and write iterators
480 iterator write_itr = read_itr;
481 ++read_itr;
482
483 // move down or skip if erased
484 while(read_itr != end_itr){
485 if(f(*read_itr)){
486 ++read_itr;
487 }else{
488 *write_itr = std::move(*read_itr);
489 ++write_itr;
490 ++read_itr;
491 }
492 }
493
494 // calculate new and erased size
495 std::size_t new_size = write_itr - this->begin();
496 std::size_t n_erased = this->end() - write_itr;
497
498 // destroy remaining
499 for(; write_itr != end_itr; ++write_itr){
500 write_itr->~T();
501 }
502 this->_size = new_size;
503 return n_erased;
504 }
