221 void promote(){
222 // promote await back into running queue
223 for(RealTimeTask* task : this->_await_bench){
224 if(!task->is_awaiting()){
225 logger::log_move(this->now(), task->name(), task->id(), "await", "run");
226 this->running_queue().push(task);
227 }
228 }
229 this->_await_bench.erase_if([](const RealTimeTask* task){return !task->is_awaiting();});
230
231 // promote waiting queue into running queue
232 while(!this->waiting_queue().empty()){
233 RealTimeTask* task = this->waiting_queue().top();
234 const TimePoint now = this->now();
235 if(task->ready_time() <= now){
236 logger::log_move(now, task->name(), task->id(), "wait", "run");
237 this->waiting_queue().pop();
238 this->running_queue().push(task);
239 }else{
240 return;
241 }
242 }
243 }
244
248 void sleep(){
249 this->_sleep_until(this->waiting_queue().top()->ready_time());
250 }
251
266 void run_next(){
267 RealTimeTask* task = this->running_queue().top_pop();
268 task->_execution_start = this->now();
269 task->resume();
270 logger::log_resume(task->_execution_start, this->now(), task->name(), task->id());
271 // re-schedule
272
273 const CoSignal signal = task->get_signal();
274 switch(signal.type()){
275 case CoSignal::Type::Await : {
276 // clear signal
277 this->_await_bench.emplace_back(task);
278 logger::log_move(this->now(), task->name(), task->id(), "resume", "await");
279 }break;
280 case CoSignal::Type::NextCycle : {
281 task->_schedule = task->next_schedule(task->_schedule, ExecutionTime{task->_execution_start, this->now()});
282 this->waiting_queue().push(task);
283 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
284 }break;
285 case CoSignal::Type::ImplicitDelay : {
286 const Duration rel_deadline = task->_schedule.deadline - task->_schedule.ready;
287 task->_schedule.ready = this->now() + embed::rounding_duration_cast<Duration>(signal.delay());
288 task->_schedule.deadline = task->_schedule.ready + rel_deadline;
289 this->waiting_queue().push(task);
290 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
291 }break;
292 case CoSignal::Type::ExplicitDelay : {
293 task->_schedule.ready = this->now() + embed::rounding_duration_cast<Duration>(signal.delay());
294 task->_schedule.deadline = task->_schedule.ready + embed::rounding_duration_cast<Duration>(signal.deadline());
295 logger::log_move(this->now(), task->name(), task->id(), "resume", "wait");
296 }break;
297 case CoSignal::Type::None : {
298 /*
299 - Nothing to do
300 - task probably finished
301 - let it die by not inserting it into lists
302 */
303 logger::log_delete(this->now(), task->name(), task->id());
304 } break;
305 default : {
306 /*
307 - Nothing to do
308 - task probably finished
309 - let it die by not inserting it into lists
310 */
311 logger::log_delete(this->now(), task->name(), task->id());
312 } break;
313 }
