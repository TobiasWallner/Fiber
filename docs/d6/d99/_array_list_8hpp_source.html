523 template<class Callable>
524 std::size_t erase_if(Callable&& f){
525 iterator read_itr = this->begin();
526 const iterator end_itr = this->end();
527
528 // search first that is true
529 for(;read_itr != end_itr; ++read_itr){
530 if(f(*read_itr)) break;
531 }
532
533 // return if it did not aplie to any
534 if(read_itr == end_itr) return 0;
535
536 // setup first read and write iterators
537 iterator write_itr = read_itr;
538 ++read_itr;
539
540 // move down or skip if erased
541 while(read_itr != end_itr){
542 if(f(*read_itr)){
543 ++read_itr;
544 }else{
545 *write_itr = std::move(*read_itr);
546 ++write_itr;
547 ++read_itr;
548 }
549 }
550
551 // calculate new and erased size
552 std::size_t new_size = std::distance(this->begin(), write_itr);
553 std::size_t n_erased = std::distance(write_itr, this->end());
554
555 // destroy remaining
556 if constexpr (!std::is_trivially_destructible_v<T>){
557 for(; write_itr != end_itr; ++write_itr){
558 write_itr->~T();
559 }
560 }
561
562 this->_size = new_size;
563 return n_erased;
564 }
