486 template<class Callable>
487 std::size_t erase_if(Callable&& f){
488 iterator read_itr = this->begin();
489 const iterator end_itr = this->end();
490
491 // search first that is true
492 for(;read_itr != end_itr; ++read_itr){
493 if(f(*read_itr)) break;
494 }
495
496 // return if it did not aplie to any
497 if(read_itr == end_itr) return 0;
498
499 // setup first read and write iterators
500 iterator write_itr = read_itr;
501 ++read_itr;
502
503 // move down or skip if erased
504 while(read_itr != end_itr){
505 if(f(*read_itr)){
506 ++read_itr;
507 }else{
508 *write_itr = std::move(*read_itr);
509 ++write_itr;
510 ++read_itr;
511 }
512 }
513
514 // calculate new and erased size
515 std::size_t new_size = std::distance(this->begin(), write_itr);
516 std::size_t n_erased = std::distance(write_itr, this->end());
517
518 // destroy remaining
519 if constexpr (!std::is_trivially_destructible_v<T>){
520 for(; write_itr != end_itr; ++write_itr){
521 write_itr->~T();
522 }
523 }
524
525 this->_size = new_size;
526 return n_erased;
527 }
